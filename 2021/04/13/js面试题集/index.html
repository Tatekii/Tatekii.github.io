function throttle(func, wait, tail,...args1) {
  if(typeof fn !== 'function' ｜｜ typeof delay !== 'number')throw new TypeError ('bad args')
  // tail = true wait结束后执行
  // tail = false 满足条件立即执行
  let timeout,pre = 0
  const _throttle = (...args2) => {
    if(tail){
      if(timeout) return
      timeout = setTimeout(()=>{
        func.apply(this,[...args1,...args2])
        clearTimeout(timeout)
        timeout = null
      },wait)
    }else{
      let now = +new Date()
      if(now-pre>wait){
        func.apply(this,[...args1,...args2])
        pre = now
      }
    }
  }
  _throttle.cancel = function(){
    clearTimeout(timeout)
    timeout = null
  }
  return _throttle
}

    let r = Promise.resolve();

function retry(fn, interval, limit) {
  return new Promise((resolve, reject) => {
    function attempt() {
      const p = Promise.resolve().then(() => fn());
      p.then(resolve).catch((err) => {
        if (limit-- <= 1) {
          reject("run out of limit");
        } else {
          const timer = setTimeout(() => {
            attempt();
            clearTimeout(timer);
          }, interval);
        }
      });
    }
    attempt();
  });
}

const isObject = (obj) => obj !== null && typeof obj === "object";

function deepClone(source, map = new WeakMap()) {
  //WeakMap 避免循环引用爆栈
  // obj.o1 = obj  // boom
  if (!isObject(source)) return source;
  if (map.has(source)) return map.get(source);
  const _constructor = source.constructor;
  if (/^(RegExp|Date)&/i.text(_constructor.name))
    return new _constructor(source);
  // 正则和日期
  const newObj =
    _constructor === Array ? source.concat() : Object.assign({}, source);
  map.set(source, newObj);
  Reflect.ownKeys(source).forEach((key) => {
    // Reflect.ownKeys 方法返回一个由目标对象自身的属性键组成的数组。它的返回值等同于Object.getOwnPropertyNames(target).concat(Object.getOwnPropertySymbols(target))。
    newObj[key] = deepClone(source[key], map);
  });
  return newObj;
}

Function.prototype._call = function (context, ...args) {
  if (typeof this !== "function") throw new TypeError("must call a function");
  context = context || window;
  const caller = Symbol("caller");
  context[caller] = this;
  let result = context[caller](...args);
  delete context[caller];
  return result;
};

Function.prototype._apply = function (context, args) {
  if (typeof this !== "function") throw new TypeError("must apply a function");
  context = context || window;
  const caller = Symbol("caller");
  context[caller] = this;
  let result = context[caller](...args);
  delete context[caller];
  return result;
};

// 返回一个原函数的拷贝，并拥有指定的this值和初始参数。
Function.prototype._bind = function (context, ...args1) {
  if (typeof this !== "function") throw new TypeError("must bind a function");
  const originFn = this;
  const notBFn = function () {};
  const boundFn = function (...args2) {
    return originFn.apply(this instanceof originFn ? this : context, [
      // 如果绑定函数被new调用
      // 则this指向new obj的this
      ...args1,
      ...args2,
    ]);
  };

  if (this.prototype) {
    notBFn.prototype = this.prototype;
  }
  boundFn.prototype = new notBFn();
  // boundFn.prototype = Object.create(this.prototype)
  // 使用notBFn间接连接boundFn和originFn原型链
  return boundFn;
};

function createIterator(array) {
  let nextIndex = 0;
  return {
    next: function () {
      return nextIndex < array.length
        ? { value: array[nextIndex++], done: false }
        : { value: undefined, done: true };
    },
    [Symbol.iterator]: function () {
      return this;
    },
  };
}

  const controller = {
    stop = false
    // 控制器
  }

    if(!constructor.stop){

    }

function bubbleSort(arr) {
  const len = arr.length;
  if (len < 2) return;
  for (i = 0; i < len - 1; i++) {
    let flag = false;
    for (j = 0; j < len - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        flag = true;
      }
    }
    if (!flag) break;
  }
  return arr;
}

const insertionSort = (arr) => {

    // 取出arr[i]插入到i之前数组中

    arr[j+1] = temp
  }
  return arr
}

function shellSort(arr, gap = 3) {
  const len = arr.length;
  let n = 1;
  while (n < len / gap) {
    n = n * gap + 1;
  }
  for (; n > 0; n = Math.floor(n / gap)) {
    for (let i = n; i < len; i++) {
      let temp = arr[i];
      let j = i - n;
      while (j >= 0 && arr[j] > temp) {
        arr[j + n] = arr[j];
        j -= n;
      }
      arr[j + n] = temp;
    }
  }
  return arr;
}

var quickSort = function (arr) {
  if (arr.length < 2) {
    return arr;
  } //一直排序到区间内只有一个数
  var pivotIndex = Math.floor(arr.length / 2); //基准位置（可任意选取）
  var pivot = arr.splice(pivotIndex, 1)[0]; //找出基准数并把它从原数组删除
  // splice()
  // arrayObject.splice(index,howmany,item1,.....,itemX)
  // splice() 方法可删除从 index 处开始的零个或多个元素，并且用参数列表中声明的一个或多个值来替换那些被删除的元素。
  // 如果从 arrayObject 中删除了元素，则返回的是含有被删除的元素的数组。
  var left = [];
  var right = [];
  for (var i = 0; i < arr.length; i++) {
    if (arr[i] < pivot) {
      left.push(arr[i]);
    } else {
      right.push(arr[i]);
    }
  }
  return [...quickSort(left), pivot, ...quickSort(right)]; //链接左数组、基准数构成的数组、右数组
