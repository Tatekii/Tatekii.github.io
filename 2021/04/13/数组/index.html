变异方法（改变原数组）

  • pop()删除最后一个元素
  • shift()删除第一个元素
  • push()添加元素至末尾
  • unshift()添加元素至首位
  • splice(start,delCount,new1,new2...)往数组索引为 start 位置删除 delCont 个元素，并从该位置加入 new 新元素
  • sort()排序
  • reverse()倒序,返回新数组

  • includes()查找是否包含元素返回Boolean,可传入起始位置
  • ...展开运算符

  • map()创建一个新数组，其结果是该数组中的每个元素都调用一个提供的函数后返回的结果

  • flatmap()使用映射函数映射每个元素，然后将结果压缩成一个新数组

  • some()方法测试是否至少有一个元素可以通过被提供的函数方法。该方法返回一个 Boolean 类型的值

  • every()检测数组所有元素是否都符合判断条件,
    如果数组中检测到有一个元素不满足, 则整个表达式返回 false,且停止遍历

  • forEach()对数组的每个元素执行一次提供的函数。*无法中途退出循环*

  • filter()返回过滤后的数组

  • find()返回数组中满足提供的测试函数的第一个元素的值。否则返回 undefined

  • findIndex()返回数组中满足提供的测试函数的第一个元素的索引。否则返回-1

  • reduce()和reduceRight()累加器,累加函数的 return 作为下一次累加的 pre 值，并可指定初始值

  • flat(deepth)按照一个可指定的深度递归遍历数组，并将所有元素与遍历到的子数组中的元素合并为一个新数组返回（会移除数组中的空项）

function flat1(arr) {
  return arr.reduce((res, item) =  {
    return res.concat(Array.isArray(item) ? flat1(item) : item);

function flat2(arr) {
  let res = [];

}

插入排序

O(n^2)

function insertSort1(arr){
  const len = arr.length
  if(len < 2) return arr
  for(let i = 1 ;i<len ;i++){
    const temp = arr[i]
    let j = i
    if(arr[j-1]>temp){
      while(arr[j-1]>temp && j>0){
        arr[j] = arr[j-1]
        j--
      }
    }
    // 将元素插入正确位置
    arr[j] = temp
  }
  return arr
}
// 交换[低效率]
function insertSort2(arr){
  const len = arr.length
  if(len < 2) return arr
  for(let i = 1 ;i < len ;i++){
    for(let j = i ;j > 0 && arr[j] >arr[j-1];j--){
      // 依次移动i指针
      // 循环比较将i位置顺序插入到i的左边
      // 依次往前冒泡到合理位置
      [arr[j],arr[j-1]] = [arr[j-1],arr[j]]
    }
  }
  return arr
}

希尔排序

平均O(n log 2n) 最坏O(n^2)

// 希尔排序相当于有间隔得进行插入排序
// 移动
function shellSort1(arr){
  const len = arr.length
  if(len<2) return arr
  let gap = Math.floor(len/2)
  for(;gap>0;gap = Math.floor(gap/2)){
    for(let i = gap;i<len;i++){
      const temp = arr[i]
      let j = i
      if(arr[j-gap]>temp){
        while(arr[j-gap]>temp && j-gap>=0){
          arr[j] = arr[j-gap]
          j-=gap
        }
      }
      arr[j] = temp
    }
  }
  return arr
}
// 交换[低效率]
function shellSort2(arr){
  const len = arr.length
  let gap = Math.floor(len/2)
  for(;gap>0;gap = Math.floor(gap/2)){
    // 不断缩小间隔的长度
    // 最终会为1
    for(let i = gap ; i<len ;i++){
      // i指针
      for(let j = i;j-gap>=0 && arr[j-gap] > arr[j];j-=gap){
        // 比较指针左边与指针位置相距gap的元素
        // 冒泡到该位置
        [arr[j],arr[j-gap]] = [arr[j-gap],arr[j]]
      }
    }
  }
  return arr
}

选择排序

时间O(n^2)

function selectionSort(arr) {
  var len = arr.length;
  var minIndex;
  for (var i = 0; i < len - 1; i++) {
    minIndex = i;
    for (var j = i + 1; j < len; j++) {
      if (arr[j] < arr[minIndex]) {
        // 寻找i右边最小的数
        minIndex = j; // 将最小数的索引保存
      }
    }
    [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]];
  }
  return arr;
}

快速排序

平均 O(n log n)

var quickSort = function (arr) {
  if (arr.length < 2) {
    return arr;
  } //一直排序到区间内只有一个数
  var pivotIndex = Math.floor(arr.length / 2); //基准位置（可任意选取）
  var pivot = arr.splice(pivotIndex, 1)[0]; //找出基准数并把它从原数组删除
  var left = [];  // 空间复杂度O(n)
  var right = [];
  for (var i = 0; i < arr.length; i++) {
    if (arr[i] < pivot) {
      left.push(arr[i]);
    } else {
      right.push(arr[i]);
    }
  }
  return [...quickSort(left), pivot, ...quickSort(right)]; 


// 不使用额外空间
  // 随机选取基准，数组分成基准左右两部分
  // 将比基准大的移动基准右边，比基准小的移动到基准左边
  // 递归分割排序
  function quickSort(arr) {
    quick(arr, 0, arr.length - 1);
    function quick(arr, left, right) {
      if(left>=right) return 
        const index = partition(arr, left, right);
        // 选出一个随机值，并把它移动到正确位置
        // 根据该位置切割递归
        if (left < index - 1) {
          //排序左边
          quick(arr, left, index - 1);
        }
        if (index < right) {
          // 排序右边
          quick(arr, index+1, right);
        }
    }
    function partition(arr, left, right) {
      const base = arr[Math.floor(Math.random() * (right - left + 1)) + left];
      // 随机选
      let L = left,
        R = right;
        // 左右指针
      while (L < R) {
        while (arr[L] < base) {
          L++;
          // 左指针向右查找比基准大
        }
        while (arr[R] > base) {
          R--;
          //右指针向左查找比基准小
        }
        if (L < R) {
          [arr[L], arr[R]] = [arr[R], arr[L]];
          //交换位置
        }
        if(arr[L]===arr[R] && L!==R){
          //重复元素
          L++
        }
      }
      //最后输出两个指针相遇位置
      //返回这个位置
      return L;
    }
    return arr
  }
  quickSort([1, 3, 4, 6, 2, 7, 5, 8]);
  //[(1, 2, 3, 4, 5, 6, 7, 8)];

堆排序

// 建立大顶堆
// 依次将末尾元素与heap[1]交换位置并调整堆，同时缩小堆的大小（排除最后的大数）
// heap[2]和heap[1]交换完成后就是升序排序
function heapSort(arr){
  const heap = [,...arr]
  let heapSize = arr.length
  buildHeap(heap,heapSize)
  // 开始交换头尾
  for(let k = heapSize;k>1;k--){
    [heap[k],heap[1]] = [heap[1],heap[k]]
    heapify(heap,--heapSize,1)
  }
  heap.shift()

  function buildHeap(heap,size){
    for(let i = Math.floor(size/2);i>=1;i--){
      heapify(heap,size,i)
    }
  }
  function heapify(heap,heapSize,i){
    while(true){
      let maxIndex = i
      if(2*i <= heapSize && heap[2*i] > heap[i]) maxIndex = 2 * i 
      if(2*i+1 <=heapSize && heap[2*i+1] > heap[maxIndex]) maxIndex = 2 * i +1
      if(maxIndex === i) break
      [heap[maxIndex],heap[i]] = [heap[i],heap[maxIndex]]
      i = maxIndex
    }
  }
}
