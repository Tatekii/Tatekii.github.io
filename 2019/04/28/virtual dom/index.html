虚拟DOM

  • 使用js模拟出来的DOM对象
  • DOM变化的对比，放在js层来做
  • 目的：提高重绘性能

js模拟

<div class=""container>
  <ul id="list">
    <li class="class">item1</li>
    <li class="class">item2</li>
    <li class="class">item3</li>
  </ul>
</div> 
  <button id="btn">改变list</button>
 //用js模拟DOM
var data={
   tag: 'ul',
   attrs: {
     id: 'list'
   },
   children: [{
     tag: 'li',
     attrs: {
       className: 'item'
     },
     children: [
       'item1'
     ]
   }, {
     tag: 'li',
     attrs: {
       className: 'item'
     },
     children: [
       'item2'
     ]
   },
   {
     tag: 'li',
     attrs: {
       className: 'item'
     },
     children: [
       'item3'
     ]
   }]
 }

 //用jquery操作

 function render (data){
   var $container = $('.container')
   $container.html('')  // <----每次都要全部清空
   var $ul = $('<ul>')
   data.children.forEach(function(item){
     $ul.append($('<'+ item.tag +'>'+ item.children[0] +'</'+ item.tag +'>'))
   })
   $container.append($ul)
 }
 render(data)
 $('#btn').click(function () {
   data.children[1].children = [Math.floor(Math.random() * 100)]
   render(data)
 })

  var snabbdom  = window.snabbdom
  //定义关键函数
  var patch = snabbdom.init([
    snabbdom_class,
    snabbdom_props,
    snabbdom_style,
    snabbdom_eventlisteners
  ])
  var h = snabbdom.h

  var container = document.getElementsByClassName('container')[0] // 注意此处的返回
  // console.log(document.getElementsByClassName('container'))
  // console.log(container)

  var vnode = h('ul#list', {}, [
    h('li.item', {}, 'item1'),
    h('li.item', {}, 'item2'),
    h('li.item', {}, 'item3')
  ])
 
  patch(container, vnode)
  
  document.getElementById('btn').addEventListener('click',function () {
    var newvnode = h('ul#list', {}, [
      h('li.item', {}, 'item1'),
      h('li.item', {}, '666666'),
      h('li.item', {}, 'item3')
    ])
    patch(vnode, newvnode) // newnode patch oldnode
  })
