  if (len < 2) return;
  for (i = 0; i < len - 1; i++) {
    let flag = false;
    for (j = 0; j < len - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];

  return arr;
}

function shuffle(arr) {
  let len = arr.length,
    i,
    temp;
  while (len) {
    i = Math.floor(Math.random() * len--);
    temp = arr[i];
    arr[i] = arr[len];
    arr[len] = temp;
  }
  return arr;
}
