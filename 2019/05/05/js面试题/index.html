// 防抖
// 多次触发最后一次执行
function debounce(fn,delay,immediate = true,context = null){
  // 有立即执行选项
  if(typeof fn !== 'function' ｜｜ typeof delay !== 'number')throw new TypeError ('bad args')
  let timer = null
  return function(...args){
    if(timer) clearTimeout(timer)
    if(immediate && !timer){
      fn.apply(context,args)
      timer = setTimeout(null,delay)
    }else{
      timer = setTimeout(()=>{
        fn.apply(context,args)
        timer = null
      },delay)
    }
  }
}


// 节流
// 连续触发限制定时间隔执行
function throttle(func,delay,tail=true,context=null){
  // tail选择满足条件就立即执行还是定时结束后执行
  if(typeof fn !== 'function' ｜｜ typeof delay !== 'number')throw new TypeError ('bad args')
  if(!tail){
    let pre = Date.now()
    return function(...args){
      let now = Date.now()
      if(now-pre>delay){
        func.apply(context,args)
        pre = now
      }
    }
  }else{
    let timer = null
    return function(...args){
      if(timer) return
      timer = setTimeout(()=>{
        func.apply(context,args)
        timer = null
      },delay)
    }
  }
}

  • 实现重试功能
function retry(fn, interval, limit) {
  return new Promise((resolve, reject) => {
    function attempt() {
      const p = Promise.resolve().then(() => fn());
      p.then(resolve).catch((err) => {
        if (limit-- <= 1) {
          reject("run out of limit");
        } else {
          const timer = setTimeout(() => {
            attempt();
            clearTimeout(timer);
          }, interval);
        }
      });
    }
    attempt();
  });
}

Function.prototype._call = function (context, ...args) {
  if (typeof this !== "function")
    throw new TypeError("must call a function");
  context = context || window;
  const caller = Symbol("caller");
  context[caller] = this;
  let result = context[caller](...args);
  delete context[caller];
  return result;
};

Function.prototype._apply = function (context, args) {
  if (typeof this !== "function")
    throw new TypeError("must apply a function");
  context = context || window;
  const caller = Symbol("caller");
  context[caller] = this;
  let result = context[caller](...args);
  delete context[caller];
  return result;
};

// 返回一个原函数的拷贝，并拥有指定的this值和初始参数。
Function.prototype._bind = function (context, ...args1) {
  if (typeof this !== "function")
    throw new TypeError("must bind a function");
  const originFn = this;
  const notBFn = function () {};
  const boundFn = function (...args2) {
    return originFn.apply(this instanceof originFn ? this : context, [
      ...args1,
      ...args2,
    ]);
  };
  // 如果绑定函数被new调用
  // 则this指向new obj的this
  if (this.prototype) {
    notBFn.prototype = this.prototype;
  }
  boundFn.prototype = new notBFn();
  // 使用notBFn间接连接boundFn和originFn原型链
  return boundFn;
};

const selfReduce = function (fn, initialIndex) {
  const arr = Array.prototype.slice.call(this);
  let startIndex;
  let res;
  if (initialIndex === undefined) {
    //寻找到非空下标
    for (let i = 0; i < arr.length; i++) {
      if (!arr.hasOwnProperty(i)) continue;
      startIndex = i;
      res = arr[i];
      break;
    }
  } else {
    startIndex = initialIndex;
  }
  for (let j = ++startIndex || 0; j < arr.length; j++) {

const selfAssign = function (target, ...source) {
  if (!target) throw "target err";
  return source.reduce((acc, cur) => {
    if (cur == null) return acc;
    isComplexType(acc) || (acc = new Object(acc));
    // ||左边为true则返回左边表达式的值得，否则返回右边表达式的值
    // 将acc变为object
    [...Object.keys(cur), ...Object.getOwnPropertySymbols(cur)].forEach(
      (key) => {
        acc[key] = cur[key];
      }
    );
    return acc;
  }, target);
};

  if (len >= 2) {
    let len1 = arr[0].length;

  }
}

  • 实现一个setTimeout

    function _setTimeout(cb,timeout,...args){
  const start = +new Date()
  let timer,now
  function loop(){
    timer = window.requestAnimationFrame(loop)
    now = +new Date()
    if(now-start>=timeout){
      cb.apply(this,args)
      window.cancelAnimationFrame(timer)
    }
  }
  window.requestAnimationFrame(loop)
}

  • 使用setTimeout模拟setInterval

    // setInterval 的问题
// 定时往事件队列中添加回调函数，如果主线程运行时间过长到大量回调超时，则有可能同时执行大量异步回调，违背使用意图
function _setInterval(fn,timeout,...args){
  const controller = {
    stop = false
    // 控制器
  }
  function interval(){
    if(!constructor.stop){
      fn.apply(this,args)
      setTimeout(interval)
    }
  }
  setTimeout(interval,timeout)
  return controller
}
