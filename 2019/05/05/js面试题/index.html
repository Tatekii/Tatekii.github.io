function retry(fn, interval, limit) {
  return new Promise((resolve, reject) => {
    function attempt() {
      const p = Promise.resolve().then(() => fn());
      p.then(resolve).catch((err) => {
        if (limit-- <= 1) {
          reject("run out of limit");
        } else {
          const timer = setTimeout(() => {
            attempt();
            clearTimeout(timer);
          }, interval);
        }
      });
    }
    attempt();
  });
}

  if (typeof this !== "function")
    throw new TypeError("must call a function");
  context = context || window;
  const caller = Symbol("caller");
  context[caller] = this;
  let result = context[caller](...args);
  delete context[caller];
  return result;
};
