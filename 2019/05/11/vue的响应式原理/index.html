Observer()

// 数据劫持的实现
function observe(obj) {
  if (!obj || typeof obj !== 'object') return
  Object.keys(obj).forEach(key => {
    defineReactive(obj, key, obj[key])
  })
}

function defineReactive(obj, key, val) {
  observe(val) //      <---------------------------递归遍历所有属性
  let dep = new Dep()
  Object.defineProperty(obj, key, {
    enumerable: true,
    configurable: true,
    get: function() {
      console.log('触发getter')
      if(Dep.target) {
        dep.addSub(Dep.target) // <------------往订阅队列里添加该订阅
      }
      return val
    },
    set: function(newVal) {
      console.log('触发setter')
        val = newVal
        dep.notify()  // <----------------通知订阅者，触发其updata视图
    }
  })
}

订阅器Dep 收集依赖关系

// 给模版添加订阅
class Dep {
  constructor() {
    this.subs = []  // 订阅队列
  }
  addSub(sub) {
    this.subs.push(sub)
  }
  notify() {
    this.subs.forEach(sub => {
      // 更新数据
      sub.update()
    })
  }
}
Dep.target = null;  //  <------------
// Dep.target每次指定订阅者之后都清空

Watcher观察者

class Watcher {
  constructor(obj, key, cb) {
    Dep.target = this  // <-------此处触发getter
    this.cb = cb
    this.obj = obj
    this.key = key
    this.value = obj[key]
    Dep.target = null
  }
  update() {
    this.value = this.obj[this.key]
    // 更新DOM
    this.cb(this.value)
  }
}

原理

  1. vue实例化，data被代理到viewmodel
  2. data被Observer绑定getter/setter
  3. 组件渲染过程中，实例触摸到的data会触发其getter,dep将该依赖收集到对应组件的watcher(订阅了这个数据)
  4. 数据改变时触发相应的setter,dep通知他的观察者watcher

VUE整个实现流程

  1. 编译模版template成render函数，执行render函数得到虚拟节点树（同时为不同类型的节点标记

  2. 将data代理到viewmodel上，并且绑定getter/setter(Object.defineProperty)

  3. 组件初次渲染视图，compile渲染html触摸到的data会触发getter收集依赖到该组件的watcher

  4. data数据改变时，触发setter,dep通知其watcher更新数据，cb将会对比新旧数据，发现数据改变了则再更新视图，这里用到了diff算法
