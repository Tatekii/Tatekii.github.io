function sort1(arr, num) {
  let initArr = arr.slice(0, num);
  let restArr = arr.splice(num);
  let sortArr = heapSort(initArr);
  for (let i = 0; i < restArr.length; i++) {
    if (restArr[i] > sortArr[0]) {
      sortArr[0] = restArr[i];
      heapSort(sortArr);
    }
  }
  return sortArr;
}
