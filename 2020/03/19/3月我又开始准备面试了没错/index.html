  let dirInfo = fs.readdirSync(entry)
  dirInfo.forEach(i=>{
    const fullPath = path.join(entry,i)
    const type = fs.statSync(fullPath)
    if(type.isDirectory()){
      dir.push(i)
      readDir(fullPath)
    }else{
      file.push(i)
    }
  })
