function bubbleSort(arr, k) {
  for (let i = 0; i < k; i++) {
    let flag = false;
    // 冒泡执行k次
    for (let j = arr.length - 1; j > 0 + i; j--) {
      if (arr[j] < arr[j - 1]) {
        [arr[j], arr[j - 1]] = [arr[j - 1], arr[j]];
        flag = true;
      }
    }
    if (!flag) break;
  }
}

function heapObj(items,k){
  const heap = [,...items.slice(0,k)]
  const heapSize = k
  buildHeap(heap,heapSize)

  // 求大值，建立最小堆
  // 后续与最小堆堆顶比较
  for(let j = k ; j < items.length ; j++){
    if(heap[1] < items[j]){
      heap[1] = items[j]
      heapify(heap,heapSize,1)
    }
  }
  return heap
  // 前k个大值
  heap.slice(1,k+1)
  //第k大的值
  heap[k]
}
function buildHeap(items,heapSize){
  if(heapSize === 1) return
  for(let i = Math.floor(heapSize/2) ; i >= 1 ; i --){
    heapify(items,heapSize,i)
  }
}
function heapify(items,heapSize,i){
  while(true){
    let minIndex = i
    if(2*i <= heapSize && items[2*i] < items[i]) minIndex = 2*i
    if(2*i+1 <= heapSize && items[2*i+1] < items[minIndex]) minIndex = 2*i+1
    if(minIndex===i) break
    [items[i],items[minIndex]] = [items[minIndex],items[i]]
    i = minIndex
  }

// 随机选取基准，数组分成基准左右两部分
// 将比基准大的移动基准右边，比基准小的移动到基准左边
// 递归分割排序
function quickSort(arr) {
  quick(arr, 0, arr.length - 1);
  function quick(arr, left, right) {
    let index;
    if (left < right) {
      //
      index = partition(arr, left, right);
      // 分离一次
      if (left < index - 1) {
        quick(arr, left, index - 1);
      }
      if (index < right) {
        quick(arr, index, right);
      }
    }
  }
  function partition(arr, left, right) {
    const base = arr[Math.floor(Math.random() * (right - left + 1)) + left];
    let L = left,
      R = right;
    while (L <= R) {
      while (arr[L] < base) {
        L++;
      }
      while (arr[R] > base) {
        R--;
      }
      if (L <= R) {
        [arr[L], arr[R]] = [arr[R], arr[L]];
        L++;
        R--;
      }
    }
    return L;
  }
}

//分割出来的索引位置，与k的索引位置比较
//第k大的值，小到大排序后就是 [nums.length - k] 索引的位置
//减少操作次数，不用完全排序一遍
function findK(nums, k) {
  return quickSelect(nums, nums.length - k);
}
function quickSelect(arr, k) {
  return quick(arr, 0, arr.length - 1, k);
}
function quick(arr, left, right, k) {
  let index;
  if (left < right) {
    index = partition(arr, left, right);
    if (k === index) {
      return arr[index];
    } else if (k < index) {
      return quick(arr, left, index - 1, k);
    } else {
      return quick(arr, index + 1, right, k);
    }
  }
  return arr[left];
}
function partition(arr, left, right) {
  let base = arr[Math.floor(Math.random() * (right - left + 1)) + left];
  let L = left,
    R = right;
  while (L < R) {
    while (arr[L] < base) {
      L++;
    }
    while (arr[R] > base) {
      R--;
    }
    if (L < R) [arr[L], arr[R]] = [arr[R], arr[L]];

    if (arr[L] === arr[R] && L !== R) L++;
  }
  return L;
}

function showMost(arr, k) {
  const map = new Map();
  const ret = [...new Set(arr)];

  arr.map((num) => {
    if (map.has(num)) {
      map.set(num, map.get(num) + 1);
    } else {
      map.set(num, 1);
    }
  });

  ret.sort((a, b) => map.get(b) - map.get(a));
  return ret.slice(0, k);
}

function showMost(arr, k) {
  const map = new Map();
  const heap = [,];
  arr.map((num) => {
    if (map.has(num)) {
      map.set(num, map.get(num) + 1);
    } else {
      map.set(num, 1);
    }
  });
  if (map.size <= k) return [...map.keys()];
  //k大于元素数，直接输出
  let times = 1;
  //记数器
  //小于k的部分直接建立堆，大于的部分开始与堆top比较
  map.forEach((value, key) => {
    if (times <= k) {
      heap.push(key);
      if (times === k) {
        //堆化
        buildHeap(heap, map, k);
      }
    } else if (map.get(heap[1]) < value) {
      // 与小顶堆顶交换
      heap[1] = key;
      heapify(heap, map, k, 1);
    }
    times++;
  });
  function buildHeap(heap, map, heapSize) {
    if (k === 1) return;
    for (let i = Math.floor(heapSize / 2); i >= 1; i--) {
      heapify(heap, map, heapSize, i);
    }
  }
  function heapify(heap, map, heapSize, i) {
    while (true) {
      let minIndex = i;
      if (2 * i <= heapSize && map.get(heap[2 * i]) < map.get(heap[minIndex]))
        minIndex = 2 * i;
      if (
        2 * i + 1 <= heapSize &&
        map.get(heap[2 * i + 1]) < map.get(heap[minIndex])
      )
        minIndex = 2 * i + 1;
      if (minIndex === i) break;
      swap(heap, minIndex, i);
      i = minIndex;
    }
  }
  heap.shift();
  return heap;
}
function swap(data, l, r) {
  let temp = data[l];
  data[l] = data[r];
  data[r] = temp;
}

function buckerSort(map,k){
  let arr = []
  let res = []
  map.forEach((value,key)=>{
    if(!arr[value]){
      arr[value] = [key]
    }else{
      arr[value].push(key)
    }
  })

  for(let i = arr.length - 1 ; i>=0 && res.length < k ; i--){
    if(arr[i]){
      res.push(...arr[i])
    }
  }
  return res
}
