function toThousand(num) {
  let [int, sec] = num.split(".");
  let res = "";
  while (int.length > 3) {
    res = "," + int.slice(-3) + res;
    int = int.slice(0, -3);
  }
  res = int + res;
  if (sec) {
    res = res + "." + sec;
  }
  return res;
}
function parseThousand(str) {
  let [int, dec] = str.split(".");
  let count = 0;
  let res = "";
  for (let i = int.length - 1; i >= 0; i--) {
    count++;
    res = int.charAt(i) + res;
    if (count % 3 === 0 && i !== 0) {
      res = "," + res;
    }
  }
  // res = res + dec? '.'+dec : ''
  if (dec) {
    res = res + "." + dec;
  }
  return res;
}
