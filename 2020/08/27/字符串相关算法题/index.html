function longestStr(str){
  let map = new Map(),maxLength = 0
  for(let i = 0; j = 0;j<str.length;j++){
    if(map.has(str[j])){
      i = Math.max(map.get(str[j])+1,i)
    }
    maxLength = Math.max(j-i+1,maxLength)
    map.set(str[j],j)
  }
  return maxLength
}
