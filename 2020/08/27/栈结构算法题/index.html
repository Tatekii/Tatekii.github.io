function MinStack(){
  this.stack = []
  this.min = null
}

MinStack.prototype.push = item => {
  if(!this.stack.length){
    this.min = item
  }else{
    this.min = Math.min(this.min,item)
  }
  this.stack.push(item)
}

MinStack.prototype.pop = () => {
  const item  = this.stack.pop()
  this.min = Math.min(...this.stack)
  return item
}

MinStack.prototype.top = () => {
  if(!this.stack.length) return null
  return this.stack[this.stack.length-1]
}

MinStack.prototype.getMin = () => {
  return this.min
}

function bracketCheck(str){
  const brackets = {
    '(':')',
    '{':'}',
    '[':']'
  }
  let stack = []
  for(let i = 0 ;i<str.length;i++){
    if(brackets[str[i]]){
      stack.push([str[i]])
    }else if(str[i] !== brackets[stack.pop()]){
      // 这个写法6⃣️啊
      return false
    }
  }
  return stack.length === 0
}
  • 删除字符串内相邻重复项
    给出由小写字母组成的字符串 S ，重复项删除操作 会选择两个相邻且相同的字母，并删除它们。

在 S 上反复执行重复项删除操作，直到无法继续删除。

在完成所有重复项删除操作后返回最终的字符串。答案保证唯一。

function delDup(S){
  let stack = [],pre = ''
  for(let c of S){
    pre = stack.pop()
    if(c !== pre){
      stack.push(pre)
      stack.push(c)
    }
  }
  return stack.join('')
}
  • 滑动窗口最大值
    // k 为窗口大小
// window存储数组索引
// 每次循环判断窗口长度
// 每次循环将arr[i]与窗口内值比较
// 保证最大值索引在窗口最左边
// 依次移动窗口，输出索引到res数组
function maxInSlideWin(arr,k){
  let res = []
  let window = []
  for(let i = 0 ; i < arr.length ; i++){
    if(i-window[0]>=k){
      window.shift()
    }
    while(arr[window[window.length-1]]<=arr[i]){
      window.pop()
    }
    window.push(i)
    if(i>=k-1){
      res.push(arr[window[0]])
    }
  }
  return res
}
