const anagrams = str => {
  if(str.length<=2) return str.length===2 ? [str , str[1] + str[0]] :str

  // 这里使用reduce的第三个参数index，正在处理的元素
  // 如果提供了initialValue，则起始索引号为0，否则从索引1起始。
  return str.split('').reduce((acc,cur,index)=>{
    let other = str.slice(0,index)+str.slice(index+1)
    // other为除了当前处理元素外的字母
    // a bcd =>
    // b cd => bcd bdc
    // c bd => cbd cdb
    // d bc => dbc dcb
    // b...
    return acc.concat(anagrams(other).map(char=>cur+char))
    // 递归至两位数数组颠倒排序,并将所有可能与当前处理元素组合
  },[])
  // 指定initialValue
}

const palindrome = str => {
  const s = str.toLowerCase().replace(/[\W_]/g,'');
  //匹配任意不是字母，数字，下划线 + _
  return s === s.split('').reverse().join('');
}
// palindrome('taco cat') -> true

function* fib(){
  let [pre,cur] = [0,1]
  for(;;){
    // 死循环的另一种写法
    // while(true)
    [pre,cur] = [cur,cur+pre]
    yield cur
  }
}
for(let x of fib()){
  if(x>100) return
  console.log(x);
